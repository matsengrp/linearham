291 void HMM::ComputeInitialForwardProbabilities() {
292  vgerm_forward_.setZero(vgerm_state_strs_.size());
293 
294  int i = 0;
295  for (auto it = vgerm_ggene_ranges_.begin(); it != vgerm_ggene_ranges_.end();
296  ++it, i++) {
297  // Obtain the (key, value) pairs from the "germline" state index map.
298  const std::string& gname = it->first;
299  int range_start, range_end;
300  std::tie(range_start, range_end) = it->second;
301 
302  // Extract the "germline" state information.
303  const GermlineGene& ggene = ggenes_.at(gname);
304  int germ_ind_start = vgerm_germ_inds_[range_start];
305 
306  // Compute the forward probability for the current "germline" state.
307  vgerm_forward_[i] = ggene.germ_ptr->gene_prob();
308  vgerm_forward_[i] *= vpadding_transition_[i];
309  vgerm_forward_[i] *= vpadding_emission_[i];
310  vgerm_forward_[i] *=
311  ggene.germ_ptr->transition()
312  .segment(germ_ind_start, range_end - range_start - 1)
313  .prod();
314  vgerm_forward_[i] *= vgerm_emission_[i];
315  }
316 
317  // Scale the forward probabilities.
318  vgerm_scaler_count_ += ScaleMatrix(vgerm_forward_);
319 };

550  std::pair<int, int>({range_start, range_end}));

559  naive_bases_.push_back(i);
560  germ_inds_.push_back(-1);
561  site_inds_.push_back(-1);

964 void FillTransition(const GermlineGene& from_ggene,
965  const GermlineGene& to_ggene, GermlineType left_gtype,
966  GermlineType right_gtype, int germ_ind_row_start,
967  int germ_ind_col_start, int site_ind_row_start,
968  int site_ind_col_start, int nti_row_start,
969  int nti_col_start, int nti_row_length, int nti_col_length,
970  int germ_row_start, int germ_col_start, int germ_row_length,
971  int germ_col_length,
972  Eigen::Ref<Eigen::MatrixXd> transition_) {
973  // Are we transitioning within the same gene?
974  // [i.e. V_i -> V_i; (N, D_i) -> (N, D_i); D_i -> D_i; (N, J_i) -> (N, J_i)]
975  if (from_ggene.germ_ptr->name() == to_ggene.germ_ptr->name()) {
976  // Are we transitioning from a NTI state in (N, D_i) or (N, J_i)?
977  if (from_ggene.type == right_gtype) {
978  // Are we in the V-D or D-J "junction" region?
979  if (nti_col_length > 0) {
980  // Fill in the N -> N transition probabilities.
981  const Eigen::MatrixXd& nti_transition =
982  (from_ggene.type == GermlineType::D)
983  ? from_ggene.DGermlinePtrCast()->nti_transition()
984  : from_ggene.JGermlinePtrCast()->nti_transition();
985  transition_.block(nti_row_start, nti_col_start, nti_row_length,
986  nti_col_length) = nti_transition;
987  }
988 
989  // Are the N -> D or N -> J transitions possible?
990  if (germ_col_length > 0) {
991  // Fill in the N -> D or N -> J transition probabilities.
992  const Eigen::MatrixXd& nti_landing_out =
993  (from_ggene.type == GermlineType::D)
994  ? from_ggene.DGermlinePtrCast()->nti_landing_out()
995  : from_ggene.JGermlinePtrCast()->nti_landing_out();
996  transition_.block(nti_row_start, germ_col_start, nti_row_length,
997  germ_col_length) =
998  nti_landing_out.block(0, germ_ind_col_start, nti_landing_out.rows(),
999  germ_col_length);
1000  }
1001  }
1002 
1003  // Are the V -> V, D -> D, or J -> J transitions possible?
1004  if (germ_row_length > 0 && germ_col_length > 0) {
1005  // Fill in the V -> V, D -> D, or J -> J transition probabilities.
1006  Eigen::Ref<Eigen::MatrixXd> transition_block = transition_.block(
1007  germ_row_start, germ_col_start, germ_row_length, germ_col_length);
1008 
1009  // Are we in the V-D or D-J "junction" region?
1010  if (germ_ind_row_start == germ_ind_col_start) {
1011  transition_block.diagonal(1) =
1012  from_ggene.germ_ptr->transition().segment(germ_ind_row_start,
1013  germ_row_length - 1);
1014  } else {
1015  transition_block.diagonal(-(germ_row_length - 1)) =
1016  from_ggene.germ_ptr->transition().diagonal(
1017  -(germ_ind_row_start + germ_row_length - 1));
1018  }
1019  }
1020  }
1021 
1022  // Are we transitioning across different genes?
1023  // [i.e. V_i -> (N, D_j) or D_i -> (N, J_j)]
1024  if (from_ggene.type == left_gtype && to_ggene.type == right_gtype) {
1025  // Are we in or transitioning to the V-D or D-J "junction" region?
1026  if (germ_row_length > 0 && nti_col_length > 0) {
1027  // Fill in the V -> N or D -> N transition probabilities.
1028  const Eigen::VectorXd& nti_landing_in =
1029  (to_ggene.type == GermlineType::D)
1030  ? to_ggene.DGermlinePtrCast()->nti_landing_in()
1031  : to_ggene.JGermlinePtrCast()->nti_landing_in();
1032  Eigen::Ref<Eigen::MatrixXd> transition_block = transition_.block(
1033  germ_row_start, nti_col_start, germ_row_length, nti_col_length);
1034 
1035  transition_block.setOnes();
1036  ColVecMatCwise(from_ggene.germ_ptr->landing_out().segment(
1037  germ_ind_row_start, germ_row_length),
1038  transition_block, transition_block);
1039  transition_block *= to_ggene.germ_ptr->gene_prob();
1040  RowVecMatCwise(nti_landing_in, transition_block, transition_block);
1041  }
1042 
1043  // Are the V -> D or D -> J transitions possible?
1044  if (germ_row_length > 0 && germ_col_length > 0) {
1045  // Is there a match region between the two genes?
1046  int match_row_diff, match_col_diff;
1047  bool match_found = false;
1048 
1049  for (int from_site_ind = site_ind_row_start;
1050  from_site_ind < site_ind_row_start + germ_row_length && !match_found;
1051  from_site_ind++) {
1052  // Can we find the start index of the potential match region?
1053  if (from_site_ind == site_ind_col_start - 1) {
1054  match_row_diff = from_site_ind - site_ind_row_start;
1055  match_col_diff = 0;
1056  match_found = true;
1057  }
1058  }
1059 
1060  for (int to_site_ind = site_ind_col_start + 1;
1061  to_site_ind < site_ind_col_start + germ_col_length && !match_found;
1062  to_site_ind++) {
1063  // Can we find the start index of the potential match region?
1064  if (site_ind_row_start == to_site_ind - 1) {
1065  match_row_diff = 0;
1066  match_col_diff = to_site_ind - site_ind_col_start;
1067  match_found = true;
1068  }
1069  }
1070 
1071  if (match_found) {
1072  // Fill in the V -> D or D -> J transition probabilities.
1073  Eigen::Ref<Eigen::MatrixXd> transition_block = transition_.block(
1074  germ_row_start + match_row_diff, germ_col_start + match_col_diff,
1075  germ_row_length - match_row_diff, germ_col_length - match_col_diff);
1076  int match_length = transition_block.diagonal().size();
1077 
1078  transition_block.diagonal().array() =
1079  from_ggene.germ_ptr->landing_out()
1080  .segment(germ_ind_row_start + match_row_diff, match_length)
1081  .array() *
1082  to_ggene.germ_ptr->gene_prob() *
1083  to_ggene.germ_ptr->landing_in()
1084  .segment(germ_ind_col_start + match_col_diff, match_length)
1085  .array();
1086  }
1087  }
1088  }
1089 };
