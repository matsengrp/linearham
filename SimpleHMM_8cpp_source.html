95 void SimpleHMM::FillGermlineEmission(
96  const std::map<std::string, std::pair<int, int>>& ggene_ranges_,
97  const std::vector<int>& germ_inds_, const std::vector<int>& site_inds_,
98  Eigen::RowVectorXd& emission_, int& scaler_count_) const {
99  emission_.setOnes(ggene_ranges_.size());
100  std::vector<int> scaler_counts(ggene_ranges_.size(), 0);
101  int max_scaler_count = 0;
102 
103  // Loop through the "germline" states and cache the associated emission
104  // probabilities.
105  int i = 0;
106  for (auto it = ggene_ranges_.begin(); it != ggene_ranges_.end(); ++it, i++) {
107  // Obtain the (key, value) pairs from the "germline" state index map.
108  const std::string& gname = it->first;
109  int range_start, range_end;
110  std::tie(range_start, range_end) = it->second;
111 
112  // Extract the "germline" state information.
113  const GermlineGene& ggene = ggenes_.at(gname);
114 
115  for (int j = range_start; j < range_end; j++) {
116  for (std::size_t k = 0; k < msa_.rows(); k++) {
117  // Is the current emitted base an unambiguous nucleotide?
118  if (msa_(k, site_inds_[j]) != alphabet_.size() - 1) {
119  emission_[i] *=
120  ggene.germ_ptr->emission()(msa_(k, site_inds_[j]), germ_inds_[j]);
121 
122  // Scale the emission probabilities.
123  scaler_counts[i] += ScaleMatrix(emission_.segment(i, 1));
124  }
125  }
126  }
127 
128  // Keep track of the maximum scaler count.
129  if (scaler_counts[i] > max_scaler_count) {
130  max_scaler_count = scaler_counts[i];
131  }
132  }
133 
134  // Make sure the emission probabilities are identically scaled.
135  scaler_count_ += max_scaler_count;
136  for (std::size_t i = 0; i < emission_.size(); i++) {
137  emission_[i] *= std::pow(SCALE_FACTOR, max_scaler_count - scaler_counts[i]);
138  }

160 void SimpleHMM::FillJunctionEmission(
161  const std::map<std::string, std::pair<int, int>>& ggene_ranges_,
162  const std::vector<int>& naive_bases_, const std::vector<int>& germ_inds_,
163  const std::vector<int>& site_inds_, std::pair<int, int> left_flexbounds,
164  std::pair<int, int> right_flexbounds, Eigen::MatrixXd& emission_) const {
165  int site_start = left_flexbounds.first;
166  int site_end = right_flexbounds.second;
167  emission_.setZero(site_end - site_start, naive_bases_.size());
168 
169  // Loop through the "junction" states and cache the associated emission
170  // probabilities.
171  for (auto it = ggene_ranges_.begin(); it != ggene_ranges_.end(); ++it) {
172  // Obtain the (key, value) pairs from the "junction" state index map.
173  const std::string& gname = it->first;
174  int range_start, range_end;
175  std::tie(range_start, range_end) = it->second;
176 
177  // Extract the "junction" state information.
178  const GermlineGene& ggene = ggenes_.at(gname);
179 
180  for (int i = range_start; i < range_end; i++) {
181  // Is the current "junction" state a NTI state?
182  if (site_inds_[i] == -1) {
183  const Eigen::MatrixXd& nti_emission =
184  (ggene.type == GermlineType::D)
185  ? ggene.DGermlinePtrCast()->nti_emission()
186  : ggene.JGermlinePtrCast()->nti_emission();
187 
188  for (int site_ind = site_start; site_ind < site_end; site_ind++) {
189  emission_(site_ind - site_start, i) = 1;
190  for (std::size_t j = 0; j < msa_.rows(); j++) {
191  // Is the current emitted base an unambiguous nucleotide?
192  if (msa_(j, site_ind) != alphabet_.size() - 1) {
193  emission_(site_ind - site_start, i) *=
194  nti_emission(msa_(j, site_ind), naive_bases_[i]);
195  }
196  }
197  }
198  } else {
199  emission_(site_inds_[i] - site_start, i) = 1;
200  for (std::size_t j = 0; j < msa_.rows(); j++) {
201  // Is the current emitted base an unambiguous nucleotide?
202  if (msa_(j, site_inds_[i]) != alphabet_.size() - 1) {
203  emission_(site_inds_[i] - site_start, i) *=
204  ggene.germ_ptr->emission()(msa_(j, site_inds_[i]),
205  germ_inds_[i]);
206  }
207  }
208  }
209  }
210  }
