123 void PhyloHMM::BuildXmsa(const std::map<std::pair<int, int>, int>& xmsa_ids) {
124  xmsa_.setConstant(msa_.rows() + 1, xmsa_ids.size(), -1);
125  xmsa_seqs_.assign(msa_.rows() + 1, "");
126 
127  // Iterate across the elements in `xmsa_ids` and incrementally build the xMSA.
128  for (auto it = xmsa_ids.begin(); it != xmsa_ids.end(); ++it) {

158 void PhyloHMM::FillGermlinePaddingEmission(
159  const std::map<std::string, std::pair<int, int>>& ggene_ranges_,
160  const Eigen::VectorXi& xmsa_inds_, Eigen::RowVectorXd& emission_,
161  int& scaler_count_) const {
162  emission_.setOnes(ggene_ranges_.size());
163  std::vector<int> scaler_counts(ggene_ranges_.size(), 0);
164  int max_scaler_count = 0;
165 
166  // Loop through the ["germline"|"padding"] states and cache the associated
167  // emission probabilities.
168  int i = 0;
169  for (auto it = ggene_ranges_.begin(); it != ggene_ranges_.end(); ++it, i++) {
170  // Obtain the start/end indices that map to the current
171  // ["germline"|"padding"] state.
172  int range_start, range_end;
173  std::tie(range_start, range_end) = it->second;
174 
175  for (int j = range_start; j < range_end; j++) {
176  emission_[i] *= xmsa_emission_[xmsa_inds_[j]];
177 
178  // Scale the emission probabilities.
179  scaler_counts[i] += ScaleMatrix(emission_.segment(i, 1));
180  }
181 
182  // Keep track of the maximum scaler count.
183  if (scaler_counts[i] > max_scaler_count) {
184  max_scaler_count = scaler_counts[i];
185  }
186  }
187 
188  // Make sure the emission probabilities are identically scaled.
189  scaler_count_ += max_scaler_count;
190  for (std::size_t i = 0; i < emission_.size(); i++) {
191  emission_[i] *= std::pow(SCALE_FACTOR, max_scaler_count - scaler_counts[i]);
192  }

220 void PhyloHMM::FillXmsaEmission() {
221  xmsa_emission_.setZero(xmsa_.cols());
222 
223  // Compute the per-site phylogenetic log-likelihoods.
224  pll_unode_t* root_node = pt::pll::GetVirtualRoot(tree_);
225  partition_->TraversalUpdate(root_node, pt::pll::TraversalType::FULL);
226  partition_->LogLikelihood(root_node, xmsa_emission_.data());
227 
228  // Apply the naive sequence correction to the phylogenetic log-likelihoods.
229  for (std::size_t i = 0; i < xmsa_emission_.size(); i++) {
230  // Is the current naive base an unambiguous nucleotide?
231  if (xmsa_(xmsa_naive_ind_, i) != alphabet_.size() - 1) {
232  double naive_prob = pi_[xmsa_(xmsa_naive_ind_, i)];
233  xmsa_emission_[i] -= std::log(naive_prob);
234  }
235  }
236 
237  xmsa_emission_.array() = xmsa_emission_.array().exp();
238 };

393 void PhyloHMM::RunPipeline(const std::string& input_path,
394  const std::string& output_path, int num_rates) {
395  // Open the RevBayes output file.
396  io::CSVReader<15, io::trim_chars<>, io::double_quote_escape<'\t', '"'>> in(
397  input_path);
398  in.read_header(io::ignore_extra_column, "Iteration", "Likelihood", "Prior",
399  "alpha", "er[1]", "er[2]", "er[3]", "er[4]", "er[5]", "er[6]",
400  "pi[1]", "pi[2]", "pi[3]", "pi[4]", "tree");
401 
402  // Initialize the output file stream.
403  std::ofstream outfile;
404  outfile.open(output_path);
405 
406  // Parse the RevBayes tree samples and compute the linearham sample
407  // information.
408  er_.assign(6, 0.0);
409  pi_.assign(4, 0.0);
410  std::string tree_str;
411  sr_.assign(num_rates, 0.0);
412 
413  int line_ind = 0;
414  while (in.read_row(iteration_, rb_loglikelihood_, prior_, alpha_, er_[0],
415  er_[1], er_[2], er_[3], er_[4], er_[5], pi_[0], pi_[1],
416  pi_[2], pi_[3], tree_str)) {
417  // Remove the node indices from the Newick tree string.
418  // Fix any missing branch lengths.
419  tree_str =
420  std::regex_replace(tree_str, std::regex("\\[\\&index=[0-9]+\\]"), "");
421  tree_ = pll_utree_parse_newick_string(tree_str.c_str());
422  pt::pll::set_missing_branch_length(tree_, EPS);
423 
424  // Calculate the site-wise rates for the current tree sample.
425  pll_compute_gamma_cats(alpha_, sr_.size(), sr_.data(),
426  PLL_GAMMA_RATES_MEAN);
427 
428  // Compute the linearham log-likelihood and sample a naive sequence.
429  InitializePhyloEmission();
430  lh_loglikelihood_ = LogLikelihood();
431  logweight_ = lh_loglikelihood_ - rb_loglikelihood_;
432  naive_sequence_ = SampleNaiveSequence();
433 
434  // Write the current tree sample to the output file.
435  if (line_ind == 0) WriteOutputHeaders(outfile);
436  WriteOutputLine(outfile);
437 
438  // Free the dynamically allocated tree structure.
439  DestroyTree();
440 
441  line_ind += 1;
442  }
443 
444  // Close the file stream.
445  outfile.close();
446 };

506  xmsa_inds_(site_ind - site_start, i));
